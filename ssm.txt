spring 属性自动注入和IOC：
	1.开启注解扫描器
	2.@Autowired 属性注解
	3.配置文件要配bean吗。service实现类，上加@service
	注：
		将@Service和@Repository放到实现类上面而不是接口类上面，
		接口只是一个规范，需要各种实现类去实现这个接口，我们要用的就是这些实用类的方法。
	
	构造型（stereotype）注解：
		开启注解扫描器后，spring会把构造型注解所标注的类,注册为bean，以后这些类对象都有spring管理。
		如@Service,@Component,@Repository,@Controller，
mybaits与spring的整合：
	它们的整合是通过spring的sessionFactory来进行的
	<!-- 配置会话工厂SqlSessionFactory -->
    <bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">
        <!-- 数据源 -->
        <property name="dataSource" ref="dataSource"/>
        <property name="mapperLocations" value="classpath:sqlmap/*Mapper.xml"/>
        <property name="typeAliasesPackage" value="com.hcg.entity" />
    </bean>
    
	<!-- 在spring容器中配置mapper的扫描器产生的动态代理对象在spring的容器中自动注册，
	bean的id就是mapper类名（首字母小写）-->
    <bean class="org.mybatis.spring.mapper.MapperScannerConfigurer">
        <!-- 指定扫描包的路径，就是mapper接口的路径，多个包中间以 半角逗号隔开   -->
        <property name="basePackage" value="com.hcg.dao"/>
        <!-- 配置sqlSessionFactoryBeanName --><!-- 配置mapper代理对象 -->
        <property name="sqlSessionFactoryBeanName" value="sqlSessionFactory"/>
    </bean>
